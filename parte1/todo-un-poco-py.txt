#7. Escribir un programa en Python que te escriba todos los números primos que hay entre el 1 y el 100
'''primos_cont = []

for a in range(2, 101):
    primo = True

    for divisor in range(2, a):
        if a % divisor == 0:
            primo = False
            break
    if primo:
        primos_cont.append(a)

print(f"Los numeros primos son: {primos_cont}")'''
